266 private void AddDataChannels()
267 {
268
269 // Creamos canales de Datos
270 RTCDataChannelInit conf = new RTCDataChannelInit() { ordered = true };
271
272 if (ImHost())
273 {
274 // Send
275 aspectRatioDataChannel = pc.CreateDataChannel("aspectRatio", conf);
276 aspectRatioDataChannel.OnOpen = async() =>
277 {
278 await Task.Delay(1000);
279 aspectRatioDataChannel.Send(BitConverter.GetBytes(videoManager.aspectRatio));
280 };
281
282 // Receive
283 pc.OnDataChannel = channel =>
284 {
285 switch (channel.Label)
286 {
287 case "clientInput":
288 clientInputDataChannel = channel;
289 clientInputDataChannel.OnMessage = bytes =>
290 {
291 var inputJson = System.Text.Encoding.UTF8.GetString(bytes);
292 inputManager.clientPosition = JsonUtility.FromJson<Vector3>(inputJson);
293 };
294 break;
295
296 case "clientAction":
297 clientActionDataChannel = channel;
298 clientActionDataChannel.OnMessage = bytes =>
299 {
300 var msg = System.Text.Encoding.UTF8.GetString(bytes);
301 switch (msg)
302 {
303 case "undo": arToolManager.UndoDrawing("Client"); break;
304 case "deleteClient": arToolManager.DeleteDrawings("Client"); break;
305 case "deleteBoth": arToolManager.DeleteDrawings("Both"); break;
306 case "ARBrush":
307 case "ARPointer":
308 case "ARMarker":
309 case "ARText":
310 arToolManager.SelectTool(PeerType.Client, msg);
311 break;
312 case "DC6B6D":
313 case "6BDC99":
314 case "6BD4DC":
315 case "FFF64A":
316 arToolManager.SelectColor(PeerType.Client, msg);
317 break;
318 }
319 };
320 break;
321 }
322 };
323 }
324 else
325
326 if (!ImHost())
327 {
328 // Send
329 clientInputDataChannel = pc.CreateDataChannel("clientInput", conf);
330 inputManager.OnClientInput += inputJson =>
331 {
332 if (clientInputDataChannel.ReadyState == RTCDataChannelState.Open)
333 {
334 clientInputDataChannel.Send(inputJson);
335 }
336 };
337
338 clientActionDataChannel = pc.CreateDataChannel("clientAction", conf);
339 clientManager.OnToolSelected += tool =>
340 {
341 if (clientActionDataChannel.ReadyState == RTCDataChannelState.Open)
342 {
343 clientActionDataChannel.Send(tool);
344 }
345 };
346 clientManager.OnUndo += () =>
347 {
348 if (clientActionDataChannel.ReadyState == RTCDataChannelState.Open)
349 {
350 clientActionDataChannel.Send("undo");
351 }
352 };
353 clientManager.OnDelete += peer =>
354 {
355 if (clientActionDataChannel.ReadyState == RTCDataChannelState.Open)
356 {
357 clientActionDataChannel.Send("delete" + peer);
358 }
359 };
360 clientManager.OnColorSelected += color =>
361 {
362 if (clientActionDataChannel.ReadyState == RTCDataChannelState.Open)
363 {
364 clientActionDataChannel.Send(color);
365 }
366 };
367
368 // Receive
369 pc.OnDataChannel = channel =>
370 {
371 switch (channel.Label)
372 {
373 case "aspectRatio":
374 aspectRatioDataChannel = channel;
375 aspectRatioDataChannel.OnMessage = bytes =>
376 {
377 videoManager.aspectRatio = BitConverter.ToSingle(bytes, 0);
378
379 videoManager.height = (int)Math.Round(videoManager.width / videoManager.aspectRatio);
380 videoManager.videoRawImage.GetComponent<AspectRatioFitter>().aspectRatio = videoManager.aspectRatio;
381 };
382 break;
383 }
384 };
385 }

392 private IEnumerator Call()
393 {
394
395 Debug.Log($"{myPeerType} - Realizando conexion");
396
397 var op = pc.CreateOffer();
398 yield return op;
399 Debug.Log($"{myPeerType} - CreateOffer: {!op.IsError}");
400 if (op.IsError) Debug.Log($"{myPeerType} - CreateOffer [Error]: {op.Error.message}");
401
402 var offer = op.Desc;
403 var op2 = pc.SetLocalDescription(ref offer);
404 yield return op2;
405 Debug.Log($"{myPeerType} - SetLocalDescription: {!op2.IsError}");
406 if (op2.IsError) Debug.Log($"{myPeerType} - SetLocalDescription [Error]: {op2.Error.message}");
407
408 Debug.Log($"{myPeerType} - Sending Offer");
409 DatabaseManager.SendMessage(RoomManager.RoomID, myPeerType, new Data { sdp = pc.LocalDescription });
410

503 private IEnumerator ReadMessageDB(Message msg)
504 {
505 //El mensaje es para mi
506 if (myPeerType != msg.peerType)
507 {
508
509 // Me envia Ice
510 if (msg.data.ice != null)
511 {
512 pc.AddIceCandidate(new RTCIceCandidate(msg.data.ice));
513 Debug.Log($"{myPeerType} - AddIceCandidate: {msg.data.ice.candidate}");
514 }
515
516 // Me envia Sdp (offer)
517 else if (msg.data.sdp.type == RTCSdpType.Offer)
518 {
519 // Guardamos offer
520 var description = msg.data.sdp;
521 var op = pc.SetRemoteDescription(ref description);
522 yield return op;
523 Debug.Log($"{myPeerType} - SetRemoteDescription: {!op.IsError}");
524
525 // Creamos Answer
526 var op2 = pc.CreateAnswer();
527 yield return op2;
528 Debug.Log($"{myPeerType} - CreateAnswer: {!op2.IsError}");
529
530
531 // La guardamos
532 var answer = op2.Desc;
533 var op3 = pc.SetLocalDescription(ref answer);
534 yield return op3;
535 Debug.Log($"{myPeerType} - SetLocalDescription: {!op3.IsError}");
536
537 // Y la enviamos
538 Debug.Log($"{myPeerType} - Sending Answer");
539 DatabaseManager.SendMessage(RoomManager.RoomID, myPeerType, new Data { sdp = pc.LocalDescription });
540 }
541
542 // Me envia Sdp (Answer)
543 else if (msg.data.sdp.type == RTCSdpType.Answer)
544 {
545 var answer = msg.data.sdp;
546 var op4 = pc.SetRemoteDescription(ref answer);
547 yield return op4;
548 Debug.Log($"{myPeerType} - SetRemoteDescription: {!op4.IsError}");
549 if (op4.IsError)
550 {
551 Debug.Log(op4.Error.message);
552 }
553
554 }
555 }
